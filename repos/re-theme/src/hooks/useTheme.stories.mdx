### Example 
```js
import React from 'react'
import { useTheme } from '@keg-hub/re-theme'

export const ThemeButton = props => {
  const theme = useTheme()
  return (
    <button style={ theme.button.default } >
      { props.children }
    </button>
  )
}
```
