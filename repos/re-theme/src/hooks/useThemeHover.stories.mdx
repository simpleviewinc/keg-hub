### Example 
```js
import React from 'react'
import { useThemeHover, useTheme } from '@keg-hub/re-theme'

export const HoverThemeButton = props => {
  const { components: { button } } = useTheme()
  const [ ref, theme ] = useThemeHover(button.default, button.hover)

  return (
    <button ref={ ref } style={ theme } >
      { props.children }
    </button>
  )
}
```
